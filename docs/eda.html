6. Relaciones entre Variables

Code 
# Crear variables numéricas derivadas para correlacionar
import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt

# --- delitos relevantes para la matriz ---
#  Seleccionamos los delitos con mayor incidencia

delitos_interes = [
    'Robo de maquinaria', 
    'Robo a negocio', 
    'Robo de vehículo automotor',
    'Homicidio culposo',
    'Secuestro', 
    'Feminicidio',
    'Lesiones culposas', 
    'Violencia familiar',
    'Homicidio doloso',
    'Abuso sexual',
    'Extorsión',
    'Daño a la propiedad'
]

# Filtrar el dataframe solo para esos delitos
df_filtrado = df[df['subtipo_delito'].isin(delitos_interes)].copy()

# --- Agrupar y pivotear ---
# Agrupamos por entidad y tipo de delito, sumando la incidencia
# (podrías agrupar también por año/mes si quisieras series de tiempo)

tabla_incidencia = df_filtrado.groupby(
    ['entidad_federativa', 'tipo_delito']
)['incidencia_delictiva'].sum().reset_index()

# Pivotear: cada delito será una columna, cada fila es un estado
tabla_pivot = tabla_incidencia.pivot(
    index='entidad_federativa', 
    columns='tipo_delito', 
    values='incidencia_delictiva'
).fillna(0)  # Los estados sin algún delito tienen 0

# --- PASO 3: Calcular la matriz de correlación ---
# Usamos correlación de Pearson (puedes probar Spearman si los datos no son lineales)
matriz_correlacion = tabla_pivot.corr(method='spearman')

# --- PASO 4: Visualizar con un heatmap ---
plt.figure(figsize=(10, 8))
sns.heatmap(
    matriz_correlacion, 
    annot=True,           # Muestra el número dentro de cada celda
    cmap='coolwarm',      # Mapa de colores: rojo=correlación positiva, azul=negativa
    center=0,             # Centro del mapa de colores en 0
    vmin=-1, vmax=1,      # Límites de la correlación
    square=True,          # Celdas cuadradas
    linewidths=0.5,       # Líneas entre celdas
    cbar_kws={"shrink": 0.8}
)
plt.title('Matriz de Correlación entre Diferentes Tipos de Delito por Estado', fontsize=14)
plt.xticks(rotation=45, ha='right')
plt.yticks(rotation=0)
plt.tight_layout()
plt.show()

import matplotlib.pyplot as plt
import seaborn as sns

orden_meses = ['Enero','Febrero','Marzo','Abril','Mayo','Junio',
               'Julio','Agosto','Septiembre','Octubre','Noviembre','Diciembre']

fig, axes = plt.subplots(2, 2, figsize=(16, 12))

# 1. Por año
por_anio = df.groupby('anio')['incidencia_delictiva'].sum().reset_index()
sns.barplot(x='anio', y='incidencia_delictiva', data=por_anio,
            ax=axes[0,0], color="#1a6fd4")
axes[0,0].set_title("Incidencia por Año", fontsize=12, fontweight='bold')
axes[0,0].set_xlabel("Año")
axes[0,0].set_ylabel("Total de casos")
axes[0,0].tick_params(axis='x', rotation=45)

# 2. Por mes
por_mes = df.groupby('mes')['incidencia_delictiva'].sum().reindex(orden_meses).reset_index()
sns.barplot(x='mes', y='incidencia_delictiva', data=por_mes,
            ax=axes[0,1], color="#c8a951")
axes[0,1].set_title("Incidencia por Mes", fontsize=12, fontweight='bold')
axes[0,1].set_xlabel("Mes")
axes[0,1].set_ylabel("Total de casos")
axes[0,1].tick_params(axis='x', rotation=45)

# 3. Por estado (top 15)
por_estado = (df.groupby('entidad_federativa')['incidencia_delictiva']
                .sum().sort_values(ascending=False).head(15).reset_index())
sns.barplot(x='incidencia_delictiva', y='entidad_federativa', data=por_estado,
            ax=axes[1,0], color="#2ecc71")
axes[1,0].set_title("Top 15 Estados con Mayor Incidencia", fontsize=12, fontweight='bold')
axes[1,0].set_xlabel("Total de casos")
axes[1,0].set_ylabel("Estado")

# 4. Por tipo de delito (top 15)
por_delito = (df.groupby('subtipo_delito')['incidencia_delictiva']
                .sum().sort_values(ascending=False).head(15).reset_index())
sns.barplot(x='incidencia_delictiva', y='subtipo_delito', data=por_delito,
            ax=axes[1,1], color="#e74c3c")
axes[1,1].set_title("Top 15 Tipos de Delito", fontsize=12, fontweight='bold')
axes[1,1].set_xlabel("Total de casos")
axes[1,1].set_ylabel("Tipo de delito")

plt.suptitle("Análisis de Incidencia Delictiva", y=1.02, fontsize=14, fontweight='bold')
plt.tight_layout()
plt.show()

## 6. Relaciones entre Variables

```{python}
#| label: correlaciones
# Crear variables numéricas derivadas para correlacionar
import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt

# --- delitos relevantes para la matriz ---
#  Seleccionamos los delitos con mayor incidencia

delitos_interes = [
    'Robo de maquinaria', 
    'Robo a negocio', 
    'Robo de vehículo automotor',
    'Homicidio culposo',
    'Secuestro', 
    'Feminicidio',
    'Lesiones culposas', 
    'Violencia familiar',
    'Homicidio doloso',
    'Abuso sexual',
    'Extorsión',
    'Daño a la propiedad'
]

# Filtrar el dataframe solo para esos delitos
df_filtrado = df[df['subtipo_delito'].isin(delitos_interes)].copy()

# --- Agrupar y pivotear ---
# Agrupamos por entidad y tipo de delito, sumando la incidencia
# (podrías agrupar también por año/mes si quisieras series de tiempo)

tabla_incidencia = df_filtrado.groupby(
    ['entidad_federativa', 'tipo_delito']
)['incidencia_delictiva'].sum().reset_index()

# Pivotear: cada delito será una columna, cada fila es un estado
tabla_pivot = tabla_incidencia.pivot(
    index='entidad_federativa', 
    columns='tipo_delito', 
    values='incidencia_delictiva'
).fillna(0)  # Los estados sin algún delito tienen 0

# --- PASO 3: Calcular la matriz de correlación ---
# Usamos correlación de Pearson (puedes probar Spearman si los datos no son lineales)
matriz_correlacion = tabla_pivot.corr(method='spearman')

# --- PASO 4: Visualizar con un heatmap ---
plt.figure(figsize=(10, 8))
sns.heatmap(
    matriz_correlacion, 
    annot=True,           # Muestra el número dentro de cada celda
    cmap='coolwarm',      # Mapa de colores: rojo=correlación positiva, azul=negativa
    center=0,             # Centro del mapa de colores en 0
    vmin=-1, vmax=1,      # Límites de la correlación
    square=True,          # Celdas cuadradas
    linewidths=0.5,       # Líneas entre celdas
    cbar_kws={"shrink": 0.8}
)
plt.title('Matriz de Correlación entre Diferentes Tipos de Delito por Estado', fontsize=14)
plt.xticks(rotation=45, ha='right')
plt.yticks(rotation=0)
plt.tight_layout()
plt.show()
```
